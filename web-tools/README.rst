=====
Bower
=====

command:
--------
    ::

    # Install bower
    $ npm install -g bower
    # Install packages
    $ bower install <package>
    # Install from bower.json
    $ bower install
    # registered package
    $ bower install jquery
    # GitHub shorthand
    $ bower install desandro/masonry
    # Git endpoint
    $ bower install git://github.com/user/package.git
    # bower install http://example.com/script.js
